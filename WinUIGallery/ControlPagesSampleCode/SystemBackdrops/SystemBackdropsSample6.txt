//Create the window
var customWindow = new SampleSystemBackdropsWindow();

//Set the list of allowed backdrops
customWindow.AllowedBackdrops = [SampleSystemBackdropsWindow.BackdropType.None,
    SampleSystemBackdropsWindow.BackdropType.Acrylic,
    SampleSystemBackdropsWindow.BackdropType.AcrylicThin,
    SampleSystemBackdropsWindow.BackdropType.Mica,
    SampleSystemBackdropsWindow.BackdropType.MicaAlt];

//Set the default backdrop
customWindow.SetBackdrop(SampleSystemBackdropsWindow.BackdropType.Mica);

//Show the window
customWindow.Activate();
