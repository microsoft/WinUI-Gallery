//Create the window
var builtInWindow = new SampleBuiltInSystemBackdropsWindow();

//Set the list of allowed backdrops
builtInWindow.AllowedBackdrops = [SampleBuiltInSystemBackdropsWindow.BackdropType.None,
    SampleBuiltInSystemBackdropsWindow.BackdropType.Acrylic,
    SampleBuiltInSystemBackdropsWindow.BackdropType.Mica,
    SampleBuiltInSystemBackdropsWindow.BackdropType.MicaAlt];

//Set the backdrop to Mica
builtInWindow.SetBackdrop(SampleBuiltInSystemBackdropsWindow.BackdropType.Mica);

//Show the window
builtInWindow.Activate();
