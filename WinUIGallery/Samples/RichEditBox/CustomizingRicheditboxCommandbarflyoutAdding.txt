--- header
Customizing RichEditBox's CommandBarFlyout - Adding 'Share'
--- xaml
<RichEditBox x:Name="REBCustom" 
    AutomationProperties.Name="editor with custom menu"
    Width="800" Height="200" 
    Loaded="REBCustom_Loaded" 
    Unloaded="REBCustom_Unloaded"/>
--- c#
private void Menu_Opening(object sender, object e)
{
    if (sender is not CommandBarFlyout myFlyout)
    {
        return;
    }

    if (myFlyout.Target == REBCustom)
    {
        AppBarButton myButton = new AppBarButton();
        myButton.Command = new StandardUICommand(StandardUICommandKind.Share);
        myFlyout.PrimaryCommands.Add(myButton);
    }
}

private void REBCustom_Loaded(object sender, RoutedEventArgs e)
{
    REBCustom.SelectionFlyout.Opening += Menu_Opening;
    REBCustom.ContextFlyout.Opening += Menu_Opening;
}

private void REBCustom_Unloaded(object sender, RoutedEventArgs e)
{
    REBCustom.SelectionFlyout.Opening -= Menu_Opening;
    REBCustom.ContextFlyout.Opening -= Menu_Opening;
}
--- c++
void MainPage::Menu_Opening(IInspectable const& sender, IInspectable const& e)
{
    auto myFlyout = sender.try_as<CommandBarFlyout>();
    if (myFlyout == nullptr)
    {
        return;
    }

    if (myFlyout.Target() == REBCustom())
    {
        AppBarButton myButton;
        myButton.Command(StandardUICommand(StandardUICommandKind::Share));
        myFlyout.PrimaryCommands().Append(myButton);
    }
}

void MainPage::REBCustom_Loaded(IInspectable const& sender, RoutedEventArgs const& e)
{
    auto selectionFlyout = REBCustom().SelectionFlyout();
    auto contextFlyout = REBCustom().ContextFlyout();
    selectionFlyout.Opening({ this, &MainPage::Menu_Opening });
    contextFlyout.Opening({ this, &MainPage::Menu_Opening });
}

void MainPage::REBCustom_Unloaded(IInspectable const& sender, RoutedEventArgs const& e)
{
    auto selectionFlyout = REBCustom().SelectionFlyout();
    auto contextFlyout = REBCustom().ContextFlyout();
    selectionFlyout.Opening(/* remove handler */);
    contextFlyout.Opening(/* remove handler */);
}
