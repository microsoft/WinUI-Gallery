--- header
Notification with AppNotification controls
--- c#
AppNotification notification = new AppNotificationBuilder()
    .AddText("Survey")
    .AddText("Please select your satisfaction level and leave a comment.")
    .AddComboBox(new AppNotificationComboBox("satisfaction")
        .AddItem("1", "Very Bad")
        .AddItem("2", "Bad")
        .AddItem("3", "Neutral")
        .AddItem("4", "Good")
        .AddItem("5", "Excellent")
        .SetSelectedItem("3"))
    .AddTextBox("comment", "Leave a comment here...","")
    .AddButton(new AppNotificationButton("Submit")
        .AddArgument("action", "submit_survey"))
    .BuildNotification();

AppNotificationManager.Default.Show(notification);
--- c++
// C++/WinRT Code
winrt::Microsoft::Windows::AppNotifications::Builder::AppNotificationComboBox comboBox(L"satisfaction");
comboBox.AddItem(L"1", L"Very Bad");
comboBox.AddItem(L"2", L"Bad");
comboBox.AddItem(L"3", L"Neutral");
comboBox.AddItem(L"4", L"Good");
comboBox.AddItem(L"5", L"Excellent");
comboBox.SetSelectedItem(L"3");

winrt::Microsoft::Windows::AppNotifications::AppNotification notification =
    winrt::Microsoft::Windows::AppNotifications::Builder::AppNotificationBuilder()
        .AddText(L"Survey")
        .AddText(L"Please select your satisfaction level and leave a comment.")
        .AddComboBox(comboBox)
        .AddTextBox(L"comment", L"Leave a comment here...", L"")
        .AddButton(winrt::Microsoft::Windows::AppNotifications::Builder::AppNotificationButton(L"Submit")
            .AddArgument(L"action", L"submit_survey"))
        .BuildNotification();

winrt::Microsoft::Windows::AppNotifications::AppNotificationManager::Default().Show(notification);
