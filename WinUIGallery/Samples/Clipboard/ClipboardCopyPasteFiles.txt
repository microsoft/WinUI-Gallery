private async void CopyFiles_Click(object sender, RoutedEventArgs args)
{
    var filePicker = new FileOpenPicker
    {
        ViewMode = PickerViewMode.List,
        FileTypeFilter = { "*" }
    };

    // Initialize the picker with the current window handle (required in WinUI 3)
    var hwnd = WinRT.Interop.WindowNative.GetWindowHandle(App.MainWindow);
    WinRT.Interop.InitializeWithWindow.Initialize(filePicker, hwnd);

    var storageItems = await filePicker.PickMultipleFilesAsync();
    if (storageItems.Count > 0)
    {
        var package = new DataPackage();
        package.SetStorageItems(storageItems);
        package.RequestedOperation = DataPackageOperation.Copy;
        Clipboard.SetContent(package);
    }
}

    picker.ViewMode(Windows::Storage::Pickers::PickerViewMode::List);
    picker.FileTypeFilter().Append(L"*");

    // Initialize the picker with the current window handle (required in WinUI 3)
    auto hwnd = WinRT::Interop::WindowNative::GetWindowHandle(App::MainWindow());

}
