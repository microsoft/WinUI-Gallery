--- header
Clipboard history and roaming options
--- c#
private void CopyWithOptions_Click(object sender, RoutedEventArgs args)
{
    var package = new DataPackage();
    package.SetText("Text with clipboard options");

    var options = new ClipboardContentOptions();
    options.IsAllowedInHistory = true;
    options.IsRoamable = true;

    Clipboard.SetContentWithOptions(package, options);
}
--- c++
void MainPage::CopyWithOptions_Click(IInspectable const& sender, RoutedEventArgs const& e)
{
    Windows::ApplicationModel::DataTransfer::DataPackage package{};
    package.SetText(L"Text with clipboard options");

    Windows::ApplicationModel::DataTransfer::ClipboardContentOptions options{};
    options.IsAllowedInHistory(true);
    options.IsRoamable(true);

    Windows::ApplicationModel::DataTransfer::Clipboard::SetContentWithOptions(package, options);
}
