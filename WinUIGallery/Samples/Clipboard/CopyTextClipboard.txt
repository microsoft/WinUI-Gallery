Copy text to the Clipboard
--- xaml
<Button Content="Copy Text to the Clipboard" Click="CopyText_Click"/>
--- c#
private void CopyText_Click(object sender, RoutedEventArgs args)
{
	var package = new DataPackage();
	package.SetText("Copy this text");
	Clipboard.SetContent(package);
}
--- c++
void MainPage::CopyText_Click(IInspectable const& sender, RoutedEventArgs const& e)
{
    Windows::ApplicationModel::DataTransfer::DataPackage package{};
    package.SetText(L"Copy this text");
    Windows::ApplicationModel::DataTransfer::Clipboard::SetContent(package);
}
