--- header
Create single threaded Multiple Top level Windows(MTW).
--- c#
// Ensure you close the child window before closing the parent window to avoid application crash.
var childWindow = new Window()
{
    ExtendsContentIntoTitleBar = true,
    SystemBackdrop = new MicaBackdrop(),
    Content = new Page()
    {
        Content = new TextBlock()
        {
            Text = "New child window!",
            HorizontalAlignment = HorizontalAlignment.Center,
            VerticalAlignment = VerticalAlignment.Center,
        },
        // Get the theme from the parent.
        RequestedTheme = this.ActualTheme,
    }
};

childWindow.AppWindow.ResizeClient(new SizeInt32(500, 500));
childWindow.Activate();
--- c++
// Ensure you close the child window before closing the parent window to avoid application crash.
Window childWindow;
childWindow.ExtendsContentIntoTitleBar(true);
childWindow.SystemBackdrop(MicaBackdrop());

TextBlock textBlock;
textBlock.Text(L"New child window!");
textBlock.HorizontalAlignment(HorizontalAlignment::Center);
textBlock.VerticalAlignment(VerticalAlignment::Center);

Page page;
page.Content(textBlock);
// Get the theme from the parent.
page.RequestedTheme(this->ActualTheme());

childWindow.Content(page);
childWindow.AppWindow().ResizeClient(SizeInt32{ 500, 500 });
childWindow.Activate();
