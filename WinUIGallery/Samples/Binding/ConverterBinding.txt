--- header
Using a converter in Binding
--- xaml
<Page...
    xmlns:c="using:YourNamespace">

    <Page.Resources>
        <!-- Declare the converter -->
        <c:EmptyStringToVisibilityConverter x:Key="EmptyStringToVisibilityConverter" />
    </Page.Resources>

    <StackPanel Spacing="8">
        <!-- Input TextBox -->
        <TextBox x:Name="InputTextBox" Header="Enter Text:" Width="300" />

        <!-- TextBlock Visibility Depends on Input Text -->
        <TextBlock Text="The input is not empty."
                    Visibility="{Binding Text, ElementName=InputTextBox, Converter={StaticResource EmptyStringToVisibilityConverter}}" />
    </StackPanel>
</Page>
--- c#
using Microsoft.UI.Xaml.Data;
using Microsoft.UI.Xaml;

namespace YourNamespace
{
    public class EmptyStringToVisibilityConverter : IValueConverter
    {
        // Visibility value when the input string is empty or null
        public Visibility EmptyValue { get; set; } = Visibility.Collapsed;

        // Visibility value when the input string is non-empty
        public Visibility NonEmptyValue { get; set; } = Visibility.Visible;

        public object Convert(object value, Type targetType, object parameter, string language)
        {
            // Return EmptyValue if the input is null or empty, otherwise NonEmptyValue
            return value is string stringValue && !string.IsNullOrEmpty(stringValue) 
                ? NonEmptyValue 
                : EmptyValue;
        }

        public object ConvertBack(object value, Type targetType, object parameter, string language)
        {
            // ConvertBack is not needed and not implemented in this case
            throw new NotImplementedException();
        }
    }
}

    EmptyStringToVisibilityConverter::EmptyStringToVisibilityConverter()
        : m_emptyValue(Microsoft::UI::Xaml::Visibility::Collapsed),
          m_nonEmptyValue(Microsoft::UI::Xaml::Visibility::Visible)
    {
    }

    Microsoft::UI::Xaml::Visibility EmptyStringToVisibilityConverter::EmptyValue()
    {
        return m_emptyValue;
    }

    void EmptyStringToVisibilityConverter::EmptyValue(Microsoft::UI::Xaml::Visibility const& value)
    {
        m_emptyValue = value;
    }

    Microsoft::UI::Xaml::Visibility EmptyStringToVisibilityConverter::NonEmptyValue()
    {
        return m_nonEmptyValue;
    }

    void EmptyStringToVisibilityConverter::NonEmptyValue(Microsoft::UI::Xaml::Visibility const& value)
    {
        m_nonEmptyValue = value;
    }

    Windows::Foundation::IInspectable EmptyStringToVisibilityConverter::Convert(
        Windows::Foundation::IInspectable const& value,
        Windows::UI::Xaml::Interop::TypeName const& /*targetType*/,
        Windows::Foundation::IInspectable const& /*parameter*/,
        hstring const& /*language*/)
    {
        // Return EmptyValue if the input is null or empty, otherwise NonEmptyValue
        if (value != nullptr)
        {
            auto str = value.as<winrt::hstring>();
            if (!str.empty())
            {
                return winrt::box_value(m_nonEmptyValue);
            }
        }
        return winrt::box_value(m_emptyValue);
    }

        Windows::Foundation::IInspectable const& /*value*/,
        Windows::UI::Xaml::Interop::TypeName const& /*targetType*/,
        Windows::Foundation::IInspectable const& /*parameter*/,
        hstring const& /*language*/)
    {
        // ConvertBack is not needed and not implemented in this case
        throw hresult_not_implemented();
    }
