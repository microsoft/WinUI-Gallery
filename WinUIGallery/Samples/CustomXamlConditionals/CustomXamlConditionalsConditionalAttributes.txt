--- header
Conditional attributes
--- xaml
<Page
    xmlns="http://schemas.microsoft.com/winfx/2006/xaml/presentation"
    xmlns:cond="using:WinUIGallery.ControlPages"
    xmlns:newExp="http://schemas.microsoft.com/winfx/2006/xaml/presentation?cond:FeatureFlagCondition(NewExperience)"
    xmlns:legacy="http://schemas.microsoft.com/winfx/2006/xaml/presentation?cond:FeatureFlagCondition(LegacyMode)">

    <!--
        At most ONE of newExp:Background and legacy:Background may evaluate to true at a time.
        If both conditions are true, the XAML parser throws at runtime.
    -->
    <Button Content="Background depends on the active flag"
            newExp:Background="Green"
            legacy:Background="Gray"
            Foreground="White" />
</Page>
--- c#
// Conditional attributes use the same setup. Make sure that, for any
// given attribute, no more than one condition can evaluate to true at
// the same time - if multiple are true the XAML parser will throw at
// runtime.
public sealed partial class MyPage : Page
{
    public MyPage()
    {
        FeatureFlagCondition.FeatureFlags["NewExperience"] = true;
        FeatureFlagCondition.FeatureFlags["LegacyMode"] = false;

        this.InitializeComponent();
    }
}
