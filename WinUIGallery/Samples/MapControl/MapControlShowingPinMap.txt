--- header
Showing a pin on the map
--- xaml
<MapControl x:Name="map1" MapServiceToken="MapServiceToken" Height="600"/>
--- c#
BasicGeoposition centerPosition = new BasicGeoposition { Latitude = 0, Longitude = 0 };
Geopoint centerPoint = new Geopoint(centerPosition);

map1.Center = centerPoint;
map1.ZoomLevel = 1;

var myLandmarks = new List<MapElement>();
BasicGeoposition position = new BasicGeoposition { Latitude = -30.034647, Longitude = -51.217659 };
Geopoint point = new Geopoint(position);

var icon = new MapIcon
{
    Location = point,
};

myLandmarks.Add(icon);

var LandmarksLayer = new MapElementsLayer
{
    MapElements = myLandmarks
};

map1.Layers.Add(LandmarksLayer);
--- c++
BasicGeoposition centerPosition{ 0, 0, 0 };
Geopoint centerPoint(centerPosition);

map1().Center(centerPoint);
map1().ZoomLevel(1);

std::vector<MapElement> myLandmarks;
BasicGeoposition position{ -30.034647, -51.217659, 0 };
Geopoint point(position);

MapIcon icon;
icon.Location(point);

myLandmarks.push_back(icon);

MapElementsLayer landmarksLayer;
landmarksLayer.MapElements(single_threaded_vector<MapElement>(std::move(myLandmarks)));

map1().Layers().Append(landmarksLayer);
