--- header
Adding tasks to the jump list
--- c#
var jumpList = await JumpList.LoadCurrentAsync();

var task = JumpListItem.CreateWithArguments("/compose", "New Message");
task.Description = "Compose a new message";
task.Logo = new Uri("ms-appx:///Assets/Tiles/AppList.targetsize-48.png");

jumpList.Items.Add(task);
await jumpList.SaveAsync();
--- c++
auto jumpList = co_await JumpList::LoadCurrentAsync();

auto task = JumpListItem::CreateWithArguments(L"/compose", L"New Message");
task.Description(L"Compose a new message");
task.Logo(Uri(L"ms-appx:///Assets/Tiles/AppList.targetsize-48.png"));

jumpList.Items().Append(task);
co_await jumpList.SaveAsync();
